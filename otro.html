cellData[cellType] : cellData[cellType][subtype]; parts.forEach(part => { const button = document.createElement('button'); button.className = 'cell-part-button'; button.textContent = part.name; button.style.left = `${part.x}%`; button.style.top = `${part.y}%`; button.style.backgroundColor = part.color || getRandomColor(); button.onclick = () => showPartInfo(part); cellImage.appendChild(button); }); } const cellData = { prokaryotic: [ /* partes de la célula procariota */ ], eukaryotic: { animal: [ /* partes de la célula animal */ ], plant: [ /* partes de la célula vegetal */ ], cytoplasm: [ // Nueva opción para la estructura del citoplasma { name: "Citosol", description: "El citosol es el fluido que llena el citoplasma y contiene organelos en suspensión.", x: 50, y: 50, color: "#add8e6" }, { name: "Citoesqueleto", description: "Es una red de fibras que da soporte a la célula y facilita el movimiento de organelos.", x: 30, y: 30, color: "#ff6347" }, { name: "Orgánulos", description: "Conjunto de estructuras especializadas que cumplen funciones específicas en la célula.", x: 70, y: 70, color: "#ffb6c1" } ] } };
